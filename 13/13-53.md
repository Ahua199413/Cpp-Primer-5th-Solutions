```c++
//移动赋值运算符版本
#include <iostream>
using std::cin; using std::cout; using std::endl; using std::cerr;
#include <string>
using std::string;

    class HasPtr{
public:
    HasPtr(const string &s = string()) : ps(new string(s)), i(0) {};
    HasPtr(const HasPtr &p) : ps(new string (*p.ps)), i(p.i) {cout << "拷贝构造函数" << endl;};    //拷贝构造函数
    HasPtr(HasPtr &&rhs) noexcept : ps(rhs.ps), i(rhs.i) { //移动构造函数
        rhs.ps = nullptr;
        cout << "移动构造函数" << endl;}
    HasPtr &operator=(const HasPtr &);  //拷贝赋值运算符
    HasPtr &operator=(HasPtr &&) noexcept ; //移动赋值运算符
    ~HasPtr(){delete ps;}
private:
    string *ps;
    int i;
};

HasPtr &
HasPtr::operator=(const HasPtr &rhs) {
    auto newp = new string(*rhs.ps);
    delete ps;
    ps = newp;
    i = rhs.i;
    cout << "拷贝赋值运算符" << endl;
    return *this;
}

HasPtr &
HasPtr::operator=(HasPtr &&rhs) noexcept {
    if (this != &rhs){
        delete ps;
        ps = rhs.ps;
        i = rhs.i;
        rhs.ps = nullptr;
    }
    cout << "移动赋值运算符" << endl;
    return *this;
}

int main(){
    HasPtr hp("test");
    HasPtr hp1, hp2;
    hp1 = std::move(hp);
}
```

```c++
//拷贝并交换版本
#include <iostream>
using std::cin; using std::cout; using std::endl; using std::cerr;
#include <string>
using std::string;

class HasPtr{
    friend void swap(HasPtr &, HasPtr &);
public:
    HasPtr(const string &s = string()) : ps(new string(s)), i(0) {};
    HasPtr(const HasPtr &p) : ps(new string (*p.ps)), i(p.i) {cout << "拷贝构造函数" << endl;};    //拷贝构造函数
    HasPtr(HasPtr &&rhs) noexcept : ps(rhs.ps), i(rhs.i) { //移动构造函数
        rhs.ps = nullptr;
        cout << "移动构造函数" << endl;};
    HasPtr &operator=(HasPtr rhs){
        swap(*this, rhs);
        return *this;
    }
    ~HasPtr(){delete ps;}
private:
    string *ps;
    int i;
};

inline
void swap(HasPtr &lhs, HasPtr &rhs) {
    using std::swap;
    swap(lhs.ps, rhs.ps);
    swap(lhs.i, rhs.i);
}

int main(){
    HasPtr hp("test");
    HasPtr hp1, hp2;
    hp1 = std::move(hp);
}
```
