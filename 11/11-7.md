```c++
#include <iostream>
using std::cin; using std::cout; using std::endl; using std::cerr;
#include <vector>
using std::vector;
#include <string>
using std::string;
#include <map>
using std::map;

using familymap = map<string, vector<string>>;

void add_nFmi(familymap &fm, const string &fm_name){
    if (fm.find(fm_name) != fm.end()){
        cerr << "This family already exist" << endl;
    } else {
        fm.insert({fm_name, vector<string>()});
    }
}

void add_Cna(familymap &fm, const string &fm_name, const string &c_name){
    auto m_iter = fm.find(fm_name);
    if (m_iter != fm.end()){
        m_iter->second.push_back(c_name);
    } else {
        cerr << "No such family named " << fm_name << " exist for its child "
             << c_name << endl;
    }
}

int main() {
    familymap fms;
    add_nFmi(fms, "Ada");
    add_nFmi(fms, "Bob");
    add_nFmi(fms, "Ada");

    add_Cna(fms, "Ada", "Alice");
    add_Cna(fms, "Bob", "Burnard");
    add_Cna(fms, "Charles", "Clark");
}
```
